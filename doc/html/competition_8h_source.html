105  void send_arm_to_zero_state(ros::Publisher & joint_trajectory_publisher) {
106  // Create a message to send.
107  trajectory_msgs::JointTrajectory msg;
108 
109  // Fill the names of the joints to be controlled.
110  // Note that the vacuum_gripper_joint is not controllable.
111  msg.joint_names.clear();
112  msg.joint_names.push_back("shoulder_pan_joint");
113  msg.joint_names.push_back("shoulder_lift_joint");
114  msg.joint_names.push_back("elbow_joint");
115  msg.joint_names.push_back("wrist_1_joint");
116  msg.joint_names.push_back("wrist_2_joint");
117  msg.joint_names.push_back("wrist_3_joint");
118  msg.joint_names.push_back("linear_arm_actuator_joint");
119  // Create one point in the trajectory.
120  msg.points.resize(1);
121  // Resize the vector to the same length as the joint names.
122  // Values are initialized to 0.
123  msg.points[0].positions.resize(msg.joint_names.size(), 0.0);
124  // How long to take getting to the point (floating point seconds).
125  msg.points[0].time_from_start = ros::Duration(0.001);
126  ROS_INFO_STREAM("Sending command:\n" << msg);
127  joint_trajectory_publisher.publish(msg);
128  }
