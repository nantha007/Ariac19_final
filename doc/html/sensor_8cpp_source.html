297  part_pose.orientation.y = camera_tf_transform_.getRotation().y();
298  part_pose.orientation.z = camera_tf_transform_.getRotation().z();
299  part_pose.orientation.w = camera_tf_transform_.getRotation().w();
300  }
301  catch (tf::TransformException &ex){
302  part_pose.orientation.w = 100.0;

356 void AriacSensorManager::LaserProfilerCallback(const sensor_msgs::LaserScan::ConstPtr & laser_msg){
357  // Get information from laser_msg
358  auto ranges = laser_msg->ranges;
359  auto max = laser_msg->angle_max;
360  auto min = laser_msg->angle_min;
361  auto increment = laser_msg->angle_increment;
362 
363  double alpha = double(min);
364  std::vector<int> heights; //list of caclulated heights off of conveyor belt
365  std::vector<double> x_positions; //list of x_positions off of center of laser profiler
366  double conv_off = .69;
367 
368  // loop through all the range data to fill height and x_poisition vectors
369  for (auto &range: ranges){
370  alpha = alpha + increment;
371  if(range<laser_msg->range_max && range>laser_msg->range_min){ //check that the range value is within acceptable range
372  int h = int((range*cos(alpha)- conv_off)*-1000); //calculate the height of the conveyor in mm
373  if (h > 3){ //range data has noise of about 2mm ignore any measurements below a threshold of 3mm
374  heights.push_back(h);
375  double x_pos = range*sin(alpha);
376  x_positions.push_back(x_pos);
377  }
378  }
379  }
380 
381  bool last_slice_was_part = cur_slice_is_part;
382  int width;
383  int avg_height;
384  int offset;
385 
386  // For Calulcating part pose
387  std::string world_frame = "/world";
388  geometry_msgs::Pose part_pose;
389  camera_tf_listener_.waitForTransform(world_frame, "/laser_profiler_1_laser_source_frame", ros::Time(0),
390  ros::Duration(3));
391  camera_tf_listener_.lookupTransform(world_frame, "/laser_profiler_1_laser_source_frame", ros::Time(0),
392  camera_tf_transform_);
393 
394  part_pose.position.x = camera_tf_transform_.getOrigin().x();
395  part_pose.position.y = camera_tf_transform_.getOrigin().y();
396  part_pose.position.z = camera_tf_transform_.getOrigin().z();
397 
398  double laser_y = part_pose.position.y;
399 
400  if (heights.size() != 0) {
401  cur_slice_is_part = 1;
402  int w = int(x_positions.back() - x_positions.front()*1000);
403  offset = int((x_positions.front() + x_positions.back())/2 *1000);
404 
405  int sum = 0;
406  for (int &h: heights)
407  sum = sum + h;
408 
409  avg_height = int(sum/heights.size());
410  }
411 
412  else
413  cur_slice_is_part = 0;
414 
415  if (cur_slice_is_part && !last_slice_was_part){
416  start_time = laser_msg->header.stamp;
417  slice_widths.clear();
418  slice_heights.clear();
419  slice_offsets.clear();
420 
421  slice_widths.push_back(width);
422  slice_heights.push_back(avg_height);
423  slice_offsets.push_back(offset);
424  }
425 
426  if (last_slice_was_part && cur_slice_is_part){
427  slice_widths.push_back(width);
428  slice_heights.push_back(avg_height);
429  slice_offsets.push_back(offset);
430  end_time = laser_msg->header.stamp;
431  }
432  else if (last_slice_was_part){
433  int bound_l = int((end_time.toSec() - start_time.toSec())*.2 *1000);
434  int bound_w = *std::max_element(slice_widths.begin(),slice_widths.end());
435  int bound_h = *std::max_element(slice_heights.begin(),slice_heights.end());
436 
437  std::string type;
438  if(bound_h < 8)
439  type = "piston_rod_part";
440  else if(bound_h >= 8 && bound_h <15)
441  type = "gear_part";
442  else if(bound_h >= 15 && bound_h <20)
443  type = "gasket_part";
444  else if(bound_h >= 20 && bound_h <30)
445  type = "disk_part";
446  else
447  type = "pulley_part";
448 
449 
450 
451  double time = start_time.toSec()+(end_time.toSec() - start_time.toSec())/2;
452 
453  int sum = 0;
454  for (int &off: slice_offsets)
455  sum = sum + off;
456 
457  double pick_y = -0.2;
458  int x_offset = int(sum/slice_offsets.size());
459  part_pose.position.x += x_offset/1000;
460  part_pose.position.y = pick_y;
461  part_pose.position.z += (double(bound_h)/1000 - conv_off);
462 
463  ROS_INFO_STREAM(bound_h);
464 
465  time = (laser_y - pick_y)/0.2 + time;
466  count_ +=1;
467  auto cur_part = std::make_tuple(type,time,part_pose,count_);
468 
469  conveyor_parts_.push_back(cur_part);
470 
471  }
